// Events
event Mirrored(uint256 indexed sourceId, address indexed follower, uint256 capital, address executor);
event Unmirrored(uint256 indexed sourceId, address indexed follower);

// Storage (graph representation chosen for O(1) presence-check + index removal)
struct Follow {
    uint256 capital;
    address executor;
    uint64 mirroredAt;
    uint256 followerIndex;   // position in followers[sourceId] for swap-and-pop unmirror
    uint256 followingIndex;  // position in following_[follower] for swap-and-pop unmirror
    bool active;
}
mapping(uint256 sourceId => mapping(address follower => Follow)) internal _follows;
mapping(uint256 sourceId => address[]) internal _followers;
mapping(address follower => uint256[]) internal _following;

// External
function mirror(uint256 sourceId, uint256 capital, address executor) external;
function unmirror(uint256 sourceId) external;
function followersOf(uint256 sourceId) external view returns (address[] memory);
function following(address user) external view returns (uint256[] memory);
function followState(uint256 sourceId, address follower) external view returns (uint256 capital, address executor, uint64 mirroredAt, bool active);
```

Validation:
- `capital == 0` → revert `ZeroCapital()` (capital-griefing prevention; documents the contract is opinionated about live follows)
- `executor == address(0)` → revert `ZeroExecutor()` (Phase 2 SequaExecutor enforces full whitelist; Phase 0 rejects zero-address only)
- Existing active follow on same (sourceId, follower) re-`mirror()` → revert `AlreadyFollowing(sourceId, follower)` (callers must `unmirror` first to change capital — keeps event stream clean)

Reentrancy (D-09): `mirror` and `unmirror` carry `nonReentrant`.

NB: Phase 0 deliberately does NOT validate that `sourceId` exists in `SourceRegistry`. The contracts are independent skeletons; cross-contract validation lands in Phase 2 when the mirror engine + executor are real. Document this in NatSpec.

    // SPDX-License-Identifier: MIT
    pragma solidity ^0.8.20;

    import {Ownable} from "@openzeppelin/contracts/access/Ownable.sol";
    import {ReentrancyGuard} from "@openzeppelin/contracts/utils/ReentrancyGuard.sol";
    import {ITradeExecutor} from "./interfaces/ITradeExecutor.sol";

    /// @title FollowRegistry
    /// @notice Phase 0 skeleton — records the on-chain follow graph and the executor each follower authorized.
    /// @dev REQ-02 + CONTEXT.md D-04 (typed executor surface), D-09 (reentrancy).
    ///      Phase 0 deliberately does NOT validate sourceId against SourceRegistry — the two skeletons are independent.
    ///      Cross-contract validation, ERC-20 allowance scope, and live executor calls land in Phase 2 (SequaExecutor.sol).
    contract FollowRegistry is Ownable, ReentrancyGuard {
        struct Follow {
            uint256 capital;
            address executor;
            uint64 mirroredAt;
            uint256 followerIndex;
            uint256 followingIndex;
            bool active;
        }

        mapping(uint256 sourceId => mapping(address follower => Follow)) internal _follows;
        mapping(uint256 sourceId => address[]) internal _followers;
        mapping(address follower => uint256[]) internal _following;

        event Mirrored(uint256 indexed sourceId, address indexed follower, uint256 capital, address executor);
        event Unmirrored(uint256 indexed sourceId, address indexed follower);

        error ZeroCapital();
        error ZeroExecutor();
        error AlreadyFollowing(uint256 sourceId, address follower);
        error NotFollowing(uint256 sourceId, address follower);

        constructor() Ownable(msg.sender) {}

        /// @notice Authorizes `executor` to mirror trades of `sourceId` for msg.sender at `capital` scale.
        /// @dev Phase 0: records intent only. Phase 2 SequaExecutor enforces the scoped ERC-20 allowance pattern (DEC-003).
        function mirror(uint256 sourceId, uint256 capital, address executor) external nonReentrant {
            if (capital == 0) revert ZeroCapital();
            if (executor == address(0)) revert ZeroExecutor();
            Follow storage f = _follows[sourceId][msg.sender];
            if (f.active) revert AlreadyFollowing(sourceId, msg.sender);

            uint256 followerIdx = _followers[sourceId].length;
            uint256 followingIdx = _following[msg.sender].length;

            _followers[sourceId].push(msg.sender);
            _following[msg.sender].push(sourceId);

            _follows[sourceId][msg.sender] = Follow({
                capital: capital,
                executor: executor,
                mirroredAt: uint64(block.timestamp),
                followerIndex: followerIdx,
                followingIndex: followingIdx,
                active: true
            });

            emit Mirrored(sourceId, msg.sender, capital, executor);
        }

        /// @notice Removes msg.sender's active follow on `sourceId`. Phase 0: graph state only.
        /// @dev Frontend (Phase 4) MUST also guide the follower to call `approve(executor, 0)` on the locked tokens
        ///      to fully revoke executor authorization (per REQ-03 plain-language custody/revocation copy).
        function unmirror(uint256 sourceId) external nonReentrant {
            Follow storage f = _follows[sourceId][msg.sender];
            if (!f.active) revert NotFollowing(sourceId, msg.sender);

            // Swap-and-pop msg.sender out of _followers[sourceId]
            uint256 idx = f.followerIndex;
            address[] storage arr = _followers[sourceId];
            uint256 last = arr.length - 1;
            if (idx != last) {
                address moved = arr[last];
                arr[idx] = moved;
                _follows[sourceId][moved].followerIndex = idx;
            }
            arr.pop();

            // Swap-and-pop sourceId out of _following[msg.sender]
            uint256 idx2 = f.followingIndex;
            uint256[] storage arr2 = _following[msg.sender];
            uint256 last2 = arr2.length - 1;
            if (idx2 != last2) {
                uint256 moved2 = arr2[last2];
                arr2[idx2] = moved2;
                _follows[moved2][msg.sender].followingIndex = idx2;
            }
            arr2.pop();

            // Clear the follow record (keep struct slot non-zero for cheap reactivation? — no, full clear is honest)
            delete _follows[sourceId][msg.sender];

            emit Unmirrored(sourceId, msg.sender);
        }

        function followersOf(uint256 sourceId) external view returns (address[] memory) {
            return _followers[sourceId];
        }

        function following(address user) external view returns (uint256[] memory) {
            return _following[user];
        }

        function followState(uint256 sourceId, address follower)
            external
            view
            returns (uint256 capital, address executor, uint64 mirroredAt, bool active)
        {
            Follow storage f = _follows[sourceId][follower];
            return (f.capital, f.executor, f.mirroredAt, f.active);
        }
    }
    ```

    // SPDX-License-Identifier: MIT
    pragma solidity ^0.8.20;

    import {Test} from "forge-std/Test.sol";
    import {FollowRegistry} from "../src/FollowRegistry.sol";

    contract FollowRegistryTest is Test {
        FollowRegistry internal registry;
        address internal alice = address(0xA11CE);
        address internal bob   = address(0xB0B);
        address internal carol = address(0xCAA01);
        address internal executor = address(0xEEEEEEEE);

        uint256 internal constant SOURCE_A = 7;
        uint256 internal constant SOURCE_B = 9;

        event Mirrored(uint256 indexed sourceId, address indexed follower, uint256 capital, address executor);
        event Unmirrored(uint256 indexed sourceId, address indexed follower);

        function setUp() public {
            registry = new FollowRegistry();
            vm.label(alice, "alice");
            vm.label(bob, "bob");
            vm.label(carol, "carol");
            vm.label(executor, "executor");
        }

        function test_mirror_recordsFollowAndEmitsMirrored() public {
            vm.expectEmit(true, true, false, true, address(registry));
            emit Mirrored(SOURCE_A, alice, 1e18, executor);
            vm.prank(alice);
            registry.mirror(SOURCE_A, 1e18, executor);

            (uint256 cap, address exec_, , bool active) = registry.followState(SOURCE_A, alice);
            assertEq(cap, 1e18);
            assertEq(exec_, executor);
            assertTrue(active);

            address[] memory followers = registry.followersOf(SOURCE_A);
            assertEq(followers.length, 1);
            assertEq(followers[0], alice);

            uint256[] memory following = registry.following(alice);
            assertEq(following.length, 1);
            assertEq(following[0], SOURCE_A);
        }

        function test_mirror_revertsOnZeroCapital() public {
            vm.expectRevert(FollowRegistry.ZeroCapital.selector);
            vm.prank(alice);
            registry.mirror(SOURCE_A, 0, executor);
        }

        function test_mirror_revertsOnZeroExecutor() public {
            vm.expectRevert(FollowRegistry.ZeroExecutor.selector);
            vm.prank(alice);
            registry.mirror(SOURCE_A, 1e18, address(0));
        }

        function test_mirror_revertsOnAlreadyFollowing() public {
            vm.prank(alice);
            registry.mirror(SOURCE_A, 1e18, executor);

            vm.expectRevert(abi.encodeWithSelector(FollowRegistry.AlreadyFollowing.selector, SOURCE_A, alice));
            vm.prank(alice);
            registry.mirror(SOURCE_A, 2e18, executor);
        }

        function test_unmirror_clearsFollowAndEmitsUnmirrored() public {
            vm.prank(alice);
            registry.mirror(SOURCE_A, 1e18, executor);

            vm.expectEmit(true, true, false, false, address(registry));
            emit Unmirrored(SOURCE_A, alice);
            vm.prank(alice);
            registry.unmirror(SOURCE_A);

            (, , , bool active) = registry.followState(SOURCE_A, alice);
            assertFalse(active);
            assertEq(registry.followersOf(SOURCE_A).length, 0);
            assertEq(registry.following(alice).length, 0);
        }

        function test_unmirror_revertsWhenNotFollowing() public {
            vm.expectRevert(abi.encodeWithSelector(FollowRegistry.NotFollowing.selector, SOURCE_A, alice));
            vm.prank(alice);
            registry.unmirror(SOURCE_A);
        }

        function test_followGraphConsistency_afterMultipleMirrorsAndUnmirrors() public {
            // alice mirrors A; bob mirrors A; carol mirrors A; alice mirrors B
            vm.prank(alice); registry.mirror(SOURCE_A, 1e18, executor);
            vm.prank(bob);   registry.mirror(SOURCE_A, 2e18, executor);
            vm.prank(carol); registry.mirror(SOURCE_A, 3e18, executor);
            vm.prank(alice); registry.mirror(SOURCE_B, 4e18, executor);

            // bob unmirrors A
            vm.prank(bob); registry.unmirror(SOURCE_A);

            // Assert followersOf(A) is {alice, carol} (order: swap-and-pop placed carol where bob was)
            address[] memory followersA = registry.followersOf(SOURCE_A);
            assertEq(followersA.length, 2);
            // After swap-and-pop of index 1 (bob) with last (carol): array becomes [alice, carol]
            assertEq(followersA[0], alice);
            assertEq(followersA[1], carol);

            // Assert followersOf(B) is {alice}
            address[] memory followersB = registry.followersOf(SOURCE_B);
            assertEq(followersB.length, 1);
            assertEq(followersB[0], alice);

            // Assert following(alice) is {A, B} (alice never unmirrored)
            uint256[] memory followingAlice = registry.following(alice);
            assertEq(followingAlice.length, 2);
            assertEq(followingAlice[0], SOURCE_A);
            assertEq(followingAlice[1], SOURCE_B);

            // Assert following(bob) is empty
            assertEq(registry.following(bob).length, 0);

            // Assert following(carol) is {A}
            uint256[] memory followingCarol = registry.following(carol);
            assertEq(followingCarol.length, 1);
            assertEq(followingCarol[0], SOURCE_A);

            // Re-mirror by bob should now succeed
            vm.prank(bob); registry.mirror(SOURCE_A, 5e18, executor);
            assertEq(registry.followersOf(SOURCE_A).length, 3);
        }
    }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| External caller → mirror | Any EOA can authorize an executor for any sourceId (intentional: follower is the trust root, not a curator) |
| Follower → executor parameter | Follower self-selects executor address; Phase 0 only rejects address(0). Phase 2 SequaExecutor enforces whitelist on the executor side |
| FollowRegistry → SourceRegistry | NONE in Phase 0 — registries are independent. Cross-validation lands in Phase 2 |
| Mirror engine → Mirrored event | Read-only consumer of the on-chain follow graph |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-00-FR-01 | Spoofing | `mirror` follower identity | mitigate | Follower IS `msg.sender` — no separately-supplied follower address means no spoofing surface. Test asserts `followersOf` returns the caller's address only |
| T-00-FR-02 | Tampering | Reentrancy on follow graph | mitigate | `nonReentrant` on `mirror` and `unmirror` (D-09); comment-stripped grep asserts ≥ 2 occurrences. Defensive baseline — Phase 2 executor wires external router calls; removing the guard later is more disruptive than keeping it now |
| T-00-FR-03 | Denial of Service | Capital-griefing (`mirror(id, 0, exec)`) | mitigate | `if (capital == 0) revert ZeroCapital();` chosen over silent-accept to keep the follow graph honest about active intent — explicit in test. Documented in NatSpec |
| T-00-FR-04 | Tampering | Phase 0 follower self-routes to a malicious `executor` address | accept (deferred to Phase 2) | Phase 0 records intent only; no funds flow. Phase 2 SequaExecutor enforces (a) whitelisted router check, (b) per-token cap, (c) slippage bounds, (d) kill switch. Documented in NatSpec. Phase 4 UI MUST surface "this contract is the official SequaExecutor" to followers before they approve |
| T-00-FR-05 | Information Disclosure | Follow graph privacy | accept | The follow graph is public by design — it's the social-discovery primitive (Sequa Project.md §3.1, REQ-10 your-follows). Documented as a feature, not a leak |
| T-00-FR-06 | Repudiation | Follower disputes a mirrored trade | mitigate | Mirrored + Unmirrored events with indexed sourceId + indexed follower give an irrefutable timeline; Phase 2 mirror engine pins every executed trade to a SignalRecorded id. Verified-on-chain badge in Phase 4 reconciles the chain to the claim |
| T-00-FR-07 | Elevation of Privilege | `Ownable` contract owner | accept | No privileged functions in Phase 0; `Ownable` is inherited solely for migration hooks (e.g., future kill-switch). Multisig transfer is hackathon-out-of-scope per CONTEXT D-08 |

Phase 2 mirror engine iterates this list per signal — engine-side pagination is a known Phase 2 concern, NOT a Phase 0 fix |
